class Box

{

private:

int* _p;


public:

Box() : _p{ nullptr } {}

Box(int p) : _p{ new int{p} } {}

Box(const Box& obj)

{

this->_p = new int;

*_p = *(obj._p);

}


Box& operator=(const Box& obj)

{

if (this->_p == nullptr)

this->_p = new int;


*_p = *(obj._p);


return *this;

}


int getP()

{

return _p != nullptr ? *_p : 0;

}


void setP(int p)

{

if (_p == nullptr)

_p = new int;

*_p = p;

}

};


void main()

{

Box a(2);

Box b;

b = a;
