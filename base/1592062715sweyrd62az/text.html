class Fraction

{

private:

int _num;

int _denom;


public:

explicit Fraction(int num = 0, int denom = 1) : _num{ num }, _denom{ denom } {}


int getNum() { return _num; }

void setNum(int num) { _num = num; }


void show()

{

cout << _num << "/" << _denom << endl;

}

};


Fraction negative(Fraction f)

{

f.setNum(-f.getNum());

return f;

}
